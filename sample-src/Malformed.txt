data List(a) of Nil() | Cons(a, List(a));

sum(xs : List(Int)) : Int = case xs
  { Nil() => 0 
  | Cons(y, ys) => y + sum(ys) };
  
append(xs : List(a), ys : List(a)) : List(a) = case xs
  {  Nil() => ys : List(b)
  |  Cons(z,zs) => Cons(z : a, z : a, append(zs,ys))
  };
